#include <type_traits>
#include <callable_traits/callable_traits.hpp>

namespace ct = callable_traits;

using expect = int;

struct foo;

template<typename T>
void test() {
    using result = ct::result_of<T>;
    static_assert(std::is_same<expect, result>{}, "");
}

int main() {

    test<int()>();
    test<int(*)()>();
    test<int(&)()>();
    test<int() const>();
    test<int(foo::*)() const>();

    auto x = []() -> int { return 0; };

    test<decltype(x)>();
}
