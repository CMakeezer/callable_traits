add_member_cv 

namespace callable_traits {

    template<typename T>
    using add_member_cv = /* implementation-defined */;

}

Constraints

T must be a Function or MemberFunctionPtr.

Behavior

  • Adds both const and volatile qualifiers to the signature of T (if not already present), preserving any pointer qualifiers, such that:
    • add_member_cv<int()> aliases int() const volatile
    • add_member_cv<int(foo::*)() volatile> aliases int(foo::*)() const volatile
    • add_member_cv<int(foo::* const &)()> aliases int(foo::* const &)() const volatile

Notes

  • Any enabled calling conventions are preserved on MemberFunctionPtr types

Example 

#include <type_traits>
#include <callable_traits/add_member_cv.hpp>

namespace ct = callable_traits;

struct foo {};

int main() {

    {
        using pmf = void(foo::*)();
        using expect = void(foo::*)() const volatile;
        using test = ct::add_member_cv<pmf>;
        static_assert(std::is_same<test, expect>::value, "");
    } {
        // add_member_cv doesn't change anything when
        // the function type is already cv-qualified.
        using pmf = void(foo::*)() const volatile &&;
        using expect = void(foo::*)() const volatile &&;
        using test = ct::add_member_cv<pmf>;
        static_assert(std::is_same<test, expect>::value, "");
    } {
        using pmf = void(foo::*)() volatile &;
        using expect = void(foo::*)() const volatile &;
        using test = ct::add_member_cv<pmf>;
        static_assert(std::is_same<test, expect>::value, "");
    } {
        // add_member_cv can also be used with "abominable"
        // function types.
        using f = void();
        using expect = void() const volatile;
        using test = ct::add_member_cv<f>;
        static_assert(std::is_same<test, expect>::value, "");
    }

// A substitution failure will occur if add_member_cv is used
// with function pointers, function references, function objects,
// or member data pointers.

}
